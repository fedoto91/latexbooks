\begin {verbatim}
/**  Postcondition: return this Student's gpa,
  *                 rounded to the nearest one 
  *                 hundredth.
  */
public double getRoundedGPA()
{  int gpaAsInt;
   gpaAsInt = (int) (gpa * 100 + 0.5);
   return gpaAsInt / 100.0;
}
\end{verbatim}

rounded to the nearest hundredth}
