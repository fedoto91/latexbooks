\RectNode(sub3)(--\underline{class Sub3}
		meth1()
		\{// concrete\}--)
\MoveToExit(3,0)

		meth1()
		\{// concrete\}--)

\ArrowSpec(H)
\Edge(sub1,super)
\Edge(sub2,super)
\Edge(sub3,sub2)
\Edge(sub4,sub2)

\EndDraw
