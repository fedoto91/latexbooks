\documentclass{article}

\usepackage{stack}

\begin{document}

\section{Simple}


\begin{stack}
  \cell{cell 1}
  \cell{cell 2}
\end{stack}

\section{Different Heights}

\begin{stack}
  \cell{default height}
  \stylepad{2}{taller}
  \stylepad{3}{even taller}
\end{stack}

\section{Different Colors}

\begin{stack}
  \cellred{cell 1}
  \cellblue{cell 2}
  \cell{cell 3}
  \cellteal{cell 4}
\end{stack}

\section{Pointers}

\begin{stack}
  \cell{cell 1} \pointer{pointer 1}
  \cell{cell 2} 
  \cell{cell 3} \pointer{pointer 2}
  \cell{cell 4}
\end{stack}

\section{Frames}
\begin{stack}

\sframe
  \cell{cell 1}
  \cell{cell 2}
  \fframe{frame 1}
  \cell{cell 3}
  \sframe
  \cell{cell 4}
  \cell{cell 5}
  \fframe{frame 2}
\end{stack}

\section{Annotation}
\begin{stack}
  \cell{cell 1}
  \cell{cell 2} \cellround{some text}
  \cell{cell 3}
\end{stack}

\section{Entire Stack}

\begin{tikzpicture}
  \entirestack{(0,0)})
  \cell{cell 1}
  \cell{cell 2}
  \cell{cell 3}
  \cell{cell 4}
  \cell{cell 5}
  \cell{cell 6}
\end{tikzpicture}

\end{document}
